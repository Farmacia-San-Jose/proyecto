package main

import (
	"log"
	"net/http"

	"github.com/gorilla/mux"
	"pharmacy_sanjose.com/go/rest-ws/handlers"
)

func main() {
	mux := mux.NewRouter()

	mux.HandleFunc("/api/v1/therepeutic_use/", handlers.GetTherepeuticUses).Methods("GET")
	mux.HandleFunc("/api/v1/therepeutic_use/{id:[0-9]+}", handlers.GetTherepeuticUse).Methods("GET")
	mux.HandleFunc("/api/v1/therepeutic_use/", handlers.CreateTherepeuticUse).Methods("POST")
	mux.HandleFunc("/api/v1/therepeutic_use/{id:[0-9]+}", handlers.UpdateTherepeuticUse).Methods("PUT")
	mux.HandleFunc("/api/v1/therepeutic_use/{id:[0-9]+}", handlers.DeleteTherepeuticUse).Methods("DELETE")

	log.Println("El servidor esta a la escucha en el puerto 5050")
	log.Fatal(http.ListenAndServe(":5050", mux))
}
